    LOCAL PROCEDURE GetCaption@1() : Text[250];
    VAR
      XBRLLine@1000 : Record 395;
    BEGIN
      IF NOT XBRLLine.GET("XBRL Taxonomy Name","XBRL Taxonomy Line No.") THEN
        EXIT('');
